```
class StackUsingQueues {
private:
    queue<int> A, B; // 使用两个队列 A 和 B
    // 单独写四个函数即可
public:
    // 入栈操作
    void push(int x) {
        // 把 A 中的所有元素先放到 B 中
        while (!A.empty()) {
            B.push(A.front());
            A.pop();
        }

        // 把新元素放到 A 中
        A.push(x);

        // 把 B 中的元素再放回 A 中
        while (!B.empty()) {
            A.push(B.front());
            B.pop();
        }
    }

    // 出栈操作
    void pop() {
        if (A.empty()) {
            cout << "Stack is empty!" << endl;
            return;
        }

        // 直接从队列 A 中移除元素（队首即为栈顶元素）
        A.pop();
    }

    // 获取栈顶元素
    int top() {
        if (A.empty()) {
            cout << "Stack is empty!" << endl;
            return -1; // 或者返回适当的错误值
        }

        // 队首即为栈顶元素
        return A.front();
    }

    // 判断栈是否为空
    bool empty() {
        return A.empty();
    }
};
```
